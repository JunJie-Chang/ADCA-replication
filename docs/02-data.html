2.3 Cleaning: from spreadsheet to respondents

I keep only valid subject-ID rows (1001–1199), which automatically drops the treatment-label rows and the trailing summary rows. One subject (ID 1060) has all 17 scenarios missing and is excluded — this reconciles the 68 rows in the file with the 67 nonhypothetical subjects the paper reports.

demo_names <- c(
  "subject","gender","age","household","child","education","income","student",
  "occupation","race","gbeef","steak_buy","other_beef","gbeef_freq","steak_freq",
  "color","brand","quality_gd","ex_fat","in_fat","price_imp","safety","juiciness",
  "flavor","tenderness","consistency","doneness","production","quality_gd2",
  "slaughter","food_safe","quality_pref","cooked","chance1","chance2","chance3",
  "chance4","chance5")
sc_names <- paste0("sc", 1:17)

raw <- read_excel("harrison.xls", sheet = "trt 10&11",
                  col_names = c(demo_names, sc_names, "blank", "session"),
                  skip = 3, col_types = "text")

df_wide <- raw |>
  mutate(subject = suppressWarnings(as.numeric(subject))) |>
  filter(!is.na(subject), subject >= 1001, subject <= 1199) |>
  filter(subject != 1060) |>                              # all scenarios missing
  mutate(treatment = factor(if_else(subject < 1100, "nonhyp", "hyp"),
                            levels = c("nonhyp", "hyp"))) |>
  mutate(across(all_of(c(setdiff(demo_names, "subject"), sc_names, "session")),
                as.numeric))

cat("Nonhypothetical subjects:", sum(df_wide$treatment == "nonhyp"), "(paper: 67)\n")
#> Nonhypothetical subjects: 67 (paper: 67)
cat("Hypothetical  subjects:",  sum(df_wide$treatment == "hyp"),    "(paper: 37)\n")
#> Hypothetical  subjects: 37 (paper: 37)

2.4 Reshape: wide → long → choice format

Two reshapes turn the respondent-level table into the estimation data:

  1. Long — one row per respondent × scenario; merge in the five prices.
  2. Choice (mlogit) format — one row per respondent × scenario × alternative, with a logical chosen flag and the alternative’s own price. Scenario 17 is excluded from estimation (it is the all-equal-price binding scenario), leaving scenarios 1–16.
df_long <- df_wide |>
  select(subject, treatment, session, all_of(sc_names)) |>
  pivot_longer(all_of(sc_names), names_to = "sc_var", values_to = "choice_code") |>
  mutate(scenario = as.integer(sub("sc", "", sc_var)),
         choice   = factor(choice_code, levels = 1:6, labels = alt_labels)) |>
  select(-sc_var, -choice_code) |>
  left_join(price_matrix, by = "scenario") |>
  arrange(subject, scenario)

df_mlogit <- df_long |>
  filter(scenario <= 16) |>                               # drop binding scenario
  crossing(alt = factor(alt_labels, levels = alt_labels)) |>
  mutate(
    chosen    = as.character(choice) == as.character(alt),
    price_alt = case_when(
      alt == "GEN" ~ GEN, alt == "GT"  ~ GT,  alt == "NAT" ~ NAT,
      alt == "CHO" ~ CHO, alt == "CAB" ~ CAB, alt == "NONE" ~ NA_real_),
    chid = paste(subject, scenario, sep = "_")) |>
  select(chid, subject, treatment, session, scenario, alt, chosen, price_alt) |>
  arrange(subject, scenario, alt)

cat("Nonhyp choice situations:",
    sum(df_mlogit$treatment == "nonhyp" & df_mlogit$alt == "GEN"), "(paper: 1072 = 67×16)\n")
#> Nonhyp choice situations: 1072 (paper: 1072 = 67×16)
cat("Hyp    choice situations:",
    sum(df_mlogit$treatment == "hyp"    & df_mlogit$alt == "GEN"), "(paper:  592 = 37×16)\n")
#> Hyp    choice situations: 592 (paper:  592 = 37×16)

  select(chid, alt, chosen, price_alt) |>
  kable(caption = "One choice situation in long format (subject 1001, scenario 1).")

## Reshape: wide → long → choice format

Two reshapes turn the respondent-level table into the estimation data:

1. **Long** — one row per *respondent × scenario*; merge in the five prices.
2. **Choice (mlogit) format** — one row per *respondent × scenario ×
   alternative*, with a logical `chosen` flag and the alternative's own price.
   **Scenario 17 is excluded from estimation** (it is the all-equal-price binding
   scenario), leaving scenarios 1–16.

```{r reshape}
df_long <- df_wide |>
  select(subject, treatment, session, all_of(sc_names)) |>
  pivot_longer(all_of(sc_names), names_to = "sc_var", values_to = "choice_code") |>
  mutate(scenario = as.integer(sub("sc", "", sc_var)),
         choice   = factor(choice_code, levels = 1:6, labels = alt_labels)) |>
  select(-sc_var, -choice_code) |>
  left_join(price_matrix, by = "scenario") |>
  arrange(subject, scenario)

df_mlogit <- df_long |>
  filter(scenario <= 16) |>                               # drop binding scenario
  crossing(alt = factor(alt_labels, levels = alt_labels)) |>
```
